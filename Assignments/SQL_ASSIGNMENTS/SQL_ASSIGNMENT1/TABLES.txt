create Database Assignment1;

use Assignment1;

--1ST TABLE DEPARTMENT

CREATE TABLE DEPARTMENT
(DEPT_NO INT PRIMARY KEY,DEPT_NAME VARCHAR(30) NOT NULL,LOCATION VARCHAR(30));

--2ND TABLE CLIENTS

CREATE TABLE CLIENTS
(CLIENT_ID INT PRIMARY KEY,CLIENT_NAME VARCHAR(30) NOT NULL,
ADDRESS VARCHAR(30),EMAIL VARCHAR(60) UNIQUE,PHONE BIGINT,BUSINESS VARCHAR(40) NOT NULL);


--3RD TABLE EMPLOYEE

CREATE TABLE EMPLOYEE
(EMP_ID INT PRIMARY KEY,EMP_NAME VARCHAR(30) NOT NULL,DESIGNATION VARCHAR(30),SALARY INT,
DEPT_NO INT FOREIGN KEY REFERENCES DEPARTMENT (DEPT_NO));


--4TH TABLE PROJECTS 

CREATE TABLE PROJECTS
(PROJECT_ID INT PRIMARY KEY , DES_CRIPTION VARCHAR(80) NOT NULL,
 STARTDATE DATE,ENDDATE DATE,ACTUAL_END_DATE DATE,
 BUDGET BIGINT CHECK(BUDGET>0),CLIENT_ID INT FOREIGN KEY REFERENCES CLIENTS(CLIENT_ID),
 CHECK(ACTUAL_END_DATE>ENDDATE));

 

--5TH TABLE EMPLOYEETASKS

 CREATE TABLE EMPLOYEETASKS
(PROJECT_ID INT FOREIGN KEY REFERENCES PROJECTS(PROJECT_ID) PRIMARY KEY,
 EMP_ID INT FOREIGN KEY REFERENCES EMPLOYEE(EMP_ID),
 STARTDATE DATE,ENDDATE DATE,TASK VARCHAR(50)NOT NULL,STATUS VARCHAR(25) NOT NULL);
