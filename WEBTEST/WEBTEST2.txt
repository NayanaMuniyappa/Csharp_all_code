<head>
    <title>Odd or Even Numbers</title>
</head>
<body>
    <h1>Odd or Even Numbers</h1>
    <label for="startNumber">Start Number: </label>
    <input type="number" id="startNumber">
    <label for="endNumber">End Number: </label>
    <input type="number" id="endNumber">
    <button onclick="checkOddEven()">Check</button>
    <ul id="results"></ul>

    <script>
        function checkOddEven() {
            const startNumber = parseInt(document.getElementById("startNumber").value);
            const endNumber = parseInt(document.getElementById("endNumber").value);
            const resultsList = document.getElementById("results");

            resultsList.innerHTML = "";

            if (isNaN(startNumber) || isNaN(endNumber)) {
                alert("Please enter valid numbers.");
                return;
            }

            for (let i = startNumber; i <= endNumber; i++) {
                const listItem = document.createElement("li");
                listItem.textContent = ${i} is ${i % 2 === 0 ? 'even' : 'odd'}.;
                resultsList.appendChild(listItem);
            }
        }
    </script>
</body>
