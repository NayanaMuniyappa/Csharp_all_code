-- DEPT TABLE

CREATE TABLE DEPT (DEPT INT PRIMARY KEY,DNAME VARCHAR(30),LOICATION VARCHAR(30));


-- EMP TABLE;

CREATE TABLE EMP(EMPNO INT,ENAME VARCHAR(30),JOB VARCHAR(20),MGR_ID INT,
HIREDATE DATE,SALARY MONEY,COMM BIGINT,
DEPTNO INT FOREIGN KEY  REFERENCES DEPT(DEPTNO),
PRIMARY KEY(EMPNO,MGR_ID));
